### Code Example

```tsx
import { IconButton } from '@zenkigen-inc/component-ui';
import { useState } from 'react';

const [isSelected, setIsSelected] = useState(false);

return (
  <IconButton
    icon="add"
    size="medium"
    variant="outline"
    isSelected={isSelected}
    onClick={() => setIsSelected(!isSelected)}
  />
);
```
