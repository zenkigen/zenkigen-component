# Color Tokens

export function ColorItem({ colorKey, value }) {
  const [copiedName, setCopiedName] = useState(false);
  const [copiedValue, setCopiedValue] = useState(false);

const handleCopyName = async () => {
try {
await navigator.clipboard.writeText(colorKey);
setCopiedName(true);
setTimeout(() => setCopiedName(false), 2000);
} catch (err) {
console.error('Failed to copy: ', err);
}
};

const handleCopyValue = async () => {
try {
await navigator.clipboard.writeText(value);
setCopiedValue(true);
setTimeout(() => setCopiedValue(false), 2000);
} catch (err) {
console.error('Failed to copy: ', err);
}
};

return (

<div className="flex flex-col items-left gap-2">
  <div className="w-full" style={{ backgroundColor: value, height: 170 }} />
  <div className="flex flex-col gap-1 text-xs text-left pr-1" style={{ minWidth: 170 }}>
    <div className="flex items-center">
      <div className="typography-label11bold">{colorKey}</div>
      <IconButton
        onClick={handleCopyName}
        size="small"
        title="カラー名をコピー"
        icon={copiedName ? 'check' : 'copy'}
        variant="text"
      />
    </div>
    <div className="flex items-center">
      <div className="typography-label11regular text-gray-500">{value}</div>
      <IconButton
        onClick={handleCopyValue}
        size="small"
        title="カラー値をコピー"
        icon={copiedValue ? 'check' : 'copy'}
        variant="text"
      />
    </div>
  </div>
</div>
); }
