### Code Example

```tsx
import { PasswordInput } from '@zenkigen-inc/component-ui';
import { useState } from 'react';

const [password, setPassword] = useState('');

return (
  <PasswordInput
    value={password}
    onChange={(e) => setPassword(e.target.value)}
    placeholder="パスワードを入力してください"
    size="medium"
  />
);
```
